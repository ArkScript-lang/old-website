35 void Compiler::compile()
36 {
37 pushHeadersPhase1();
38
39 if (m_debug >= 1)
40 std::cout << "Compiling\n";
41
42 // gather symbols, values, and start to create code segments
43 m_code_pages.emplace_back(); // create empty page
44
45 _compile(m_optimizer.ast(), 0);
46 // throw an error on undefined symbol uses
47 checkForUndefinedSymbol();
48
49 pushHeadersPhase2();
50
51 // start code segments
52 for (auto page : m_code_pages)
53 {
54 m_bytecode.push_back(Instruction::CODE_SEGMENT_START);
55 // push number of elements
56 if (!page.size())
57 {
58 pushNumber(0x01);
59 m_bytecode.push_back(Instruction::HALT);
60 return;
61 }
62 pushNumber(static_cast<uint16_t>(page.size() + 1));
63
64 for (auto inst : page)
65 m_bytecode.push_back(inst);
66 // just in case we got too far, always add a HALT to be sure the
67 // VM won't do anything crazy
68 m_bytecode.push_back(Instruction::HALT);
69 }
70
71 if (!m_code_pages.size())
72 {
73 m_bytecode.push_back(Instruction::CODE_SEGMENT_START);
74 pushNumber(static_cast<uint16_t>(1));
75 m_bytecode.push_back(Instruction::HALT);
76 }
77
78 constexpr std::size_t header_size = 18;
79
80 // generate a hash of the tables + bytecode
81 std::vector<unsigned char> hash(picosha2::k_digest_size);
82 picosha2::hash256(m_bytecode.begin() + header_size, m_bytecode.end(), hash);
83 m_bytecode.insert(m_bytecode.begin() + header_size, hash.begin(), hash.end());

101 void Compiler::pushHeadersPhase1() noexcept
102 {
103 /*
104 Generating headers:
105 - lang name (to be sure we are executing an ArkScript file)
106 on 4 bytes (ark + padding)
107 - version (major: 2 bytes, minor: 2 bytes, patch: 2 bytes)
108 - timestamp (8 bytes, unix format)
109 */
110
111 m_bytecode.push_back('a');
112 m_bytecode.push_back('r');
113 m_bytecode.push_back('k');
114 m_bytecode.push_back(Instruction::NOP);
115
116 // push version
117 pushNumber(ARK_VERSION_MAJOR);
118 pushNumber(ARK_VERSION_MINOR);
119 pushNumber(ARK_VERSION_PATCH);
120
121 // push timestamp
122 unsigned long long timestamp = std::chrono::duration_cast<std::chrono::seconds>(
123 std::chrono::system_clock::now().time_since_epoch())
124 .count();
125 for (char c = 0; c < 8; c++)
126 {
127 unsigned d = 56 - 8 * c;
128 uint8_t b = (timestamp & (0xff << d)) >> d;
129 m_bytecode.push_back(b);
130 }
131
132 if (m_debug >= 1)
133 std::cout << "Timestamp: " << timestamp << '\n';
134 }

136 void Compiler::pushHeadersPhase2()
137 {
138 /*
139 - symbols table
140 + elements
141 - values table header
142 + elements
143 */
144
145 m_bytecode.push_back(Instruction::SYM_TABLE_START);
146 // push size
147 pushNumber(static_cast<uint16_t>(m_symbols.size()));
148 // push elements
149 for (auto sym : m_symbols)
150 {
151 // push the string, null terminated
152 std::string s = sym.string();
153 for (std::size_t i = 0, size = s.size(); i < size; ++i)
154 m_bytecode.push_back(s[i]);
155 m_bytecode.push_back(Instruction::NOP);
156 }
157
158 // values table
159 m_bytecode.push_back(Instruction::VAL_TABLE_START);
160 // push size
161 pushNumber(static_cast<uint16_t>(m_values.size()));
162 // push elements (separated with 0x00)
163 for (auto val : m_values)
164 {
165 if (val.type == CValueType::Number)
166 {
167 m_bytecode.push_back(Instruction::NUMBER_TYPE);
168 auto n = std::get<double>(val.value);
169 std::string t = std::to_string(n);
170 for (std::size_t i = 0, size = t.size(); i < size; ++i)
171 m_bytecode.push_back(t[i]);
172 }
173 else if (val.type == CValueType::String)
174 {
175 m_bytecode.push_back(Instruction::STRING_TYPE);
176 std::string t = std::get<std::string>(val.value);
177 for (std::size_t i = 0, size = t.size(); i < size; ++i)
178 m_bytecode.push_back(t[i]);
179 }
180 else if (val.type == CValueType::PageAddr)
181 {
182 m_bytecode.push_back(Instruction::FUNC_TYPE);
183 pushNumber(static_cast<uint16_t>(std::get<std::size_t>(val.value)));
184 }
185 else
186 throw Ark::CompilationError("trying to put a value in the value table, but the type isn't handled.\nCertainly a logic problem in the compiler source code");
187
188 m_bytecode.push_back(Instruction::NOP);
189 }

192 void Compiler::_compile(const Node& x, int p)
193 {
194 if (m_debug >= 4)
195 std::cout << x << '\n';
196
197 // register symbols
198 if (x.nodeType() == NodeType::Symbol)
199 {
200 compileSymbol(x, p);
201 return;
202 }
203 else if (x.nodeType() == NodeType::GetField)
204 {
205 std::string name = x.string();
206 // 'name' shouldn't be a builtin/operator, we can use it as-is
207 std::size_t i = addSymbol(x);
208
209 page(p).emplace_back(Instruction::GET_FIELD);
210 pushNumber(static_cast<uint16_t>(i), &page(p));
211
212 return;
213 }
214 // register values
215 else if (x.nodeType() == NodeType::String || x.nodeType() == NodeType::Number)
216 {
217 std::size_t i = addValue(x);
218
219 page(p).emplace_back(Instruction::LOAD_CONST);
220 pushNumber(static_cast<uint16_t>(i), &page(p));
221
222 return;
223 }
224 // empty code block should be nil
225 else if (x.constList().empty())
226 {
227 auto it_builtin = isBuiltin("nil");
228 page(p).emplace_back(Instruction::BUILTIN);
229 pushNumber(static_cast<uint16_t>(it_builtin.value()), &page(p));
230 return;
231 }
232 // specific instructions
233 else if (auto c0 = x.constList()[0]; c0.nodeType() == NodeType::Symbol && isSpecific(c0.string()).has_value())
234 {
235 compileSpecific(c0, x, p);
236 return;
237 }
238 // registering structures
239 else if (x.constList()[0].nodeType() == NodeType::Keyword)
240 {
241 Keyword n = x.constList()[0].keyword();
242
243 if (n == Keyword::If)
244 compileIf(x, p);
245 else if (n == Keyword::Set)
246 compileSet(x, p);
247 else if (n == Keyword::Let || n == Keyword::Mut)
248 compileLetMut(n, x, p);
249 else if (n == Keyword::Fun)
250 compileFunction(x, p);
251 else if (n == Keyword::Begin)
252 {
253 for (std::size_t i = 1, size = x.constList().size(); i < size; ++i)
254 _compile(x.constList()[i], p);
255 }
256 else if (n == Keyword::While)
257 compileWhile(x, p);
258 else if (n == Keyword::Import)
259 compilePluginImport(x, p);
260 else if (n == Keyword::Quote)
261 compileQuote(x, p);
262 else if (n == Keyword::Del)
263 compileDel(x, p);
264
265 return;
266 }
267
268 // if we are here, we should have a function name
269 // push arguments first, then function name, then call it
270 handleCalls(x, p);
271 return;
272 }

294 void Compiler::compileSpecific(const Node& c0, const Node& x, int p)
295 {
296 std::string name = c0.string();
297 Instruction inst = isSpecific(name).value();
298
299 // length of at least 1 since we got a symbol name
300 uint16_t argc = countArkObjects(x.constList()) - 1;
301 // error, can not use append/concat/pop (and their in place versions) with a <2 length argument list
302 if (argc < 2 && inst != Instruction::LIST)
303 throw Ark::CompilationError("can not use " + name + " with less than 2 arguments");
304
305 // compile arguments in reverse order
306 for (uint16_t i = x.constList().size() - 1; i > 0; --i)
307 {
308 uint16_t j = i;
309 while (x.constList()[j].nodeType() == NodeType::GetField)
310 --j;
311 uint16_t diff = i - j;
312 while (j < i)
313 {
314 _compile(x.constList()[j], p);
315 ++j;
316 }
317 _compile(x.constList()[i], p);
318 i -= diff;
319 }
320
321 // put inst and number of arguments
322 page(p).emplace_back(inst);
323 pushSpecificInstArgc(inst, argc, p);
324 }

327 {
328 // compile condition
329 _compile(x.constList()[1], p);
330 // jump only if needed to the x.list()[2] part
331 page(p).emplace_back(Instruction::POP_JUMP_IF_TRUE);
332 std::size_t jump_to_if_pos = page(p).size();

344 page(p)[jump_to_if_pos + 1] = static_cast<uint16_t>(page(p).size()) & 0x00ff;

352 void Compiler::compileFunction(const Node& x, int p)
353 {
354 // capture, if needed
355 for (auto it = x.constList()[1].constList().begin(), it_end = x.constList()[1].constList().end(); it != it_end; ++it)
356 {
357 if (it->nodeType() == NodeType::Capture)
358 {
359 // first check that the capture is a defined symbol
360 if (std::find(m_defined_symbols.begin(), m_defined_symbols.end(), it->string()) == m_defined_symbols.end())
361 {
362 // we didn't find it in the defined symbol list, thus we can't capture it
363 throwCompilerError("Can not capture " + it->string() + " because it is referencing an unbound variable.", *it);
364 }
365 page(p).emplace_back(Instruction::CAPTURE);
366 addDefinedSymbol(it->string());
367 std::size_t var_id = addSymbol(*it);
368 pushNumber(static_cast<uint16_t>(var_id), &(page(p)));
369 }
370 }
371 // create new page for function body
372 m_code_pages.emplace_back();
373 std::size_t page_id = m_code_pages.size() - 1;
374 // load value on the stack
375 page(p).emplace_back(Instruction::LOAD_CONST);
376 std::size_t id = addValue(page_id); // save page_id into the constants table as PageAddr
377 pushNumber(static_cast<uint16_t>(id), &page(p));
378 // pushing arguments from the stack into variables in the new scope
379 for (auto it = x.constList()[1].constList().begin(), it_end = x.constList()[1].constList().end(); it != it_end; ++it)
380 {
381 if (it->nodeType() == NodeType::Symbol)
382 {
383 page(page_id).emplace_back(Instruction::MUT);
384 std::size_t var_id = addSymbol(*it);
385 addDefinedSymbol(it->string());
386 pushNumber(static_cast<uint16_t>(var_id), &(page(page_id)));
387 }
388 }
389 // push body of the function
390 _compile(x.constList()[2], page_id);
391 // return last value on the stack
392 page(page_id).emplace_back(Instruction::RET);
393 }

490 void Compiler::handleCalls(const Node& x, int p)
491 {
492 m_temp_pages.emplace_back();
493 int proc_page = -static_cast<int>(m_temp_pages.size());
494 _compile(x.constList()[0], proc_page); // storing proc
495
496 // trying to handle chained closure.field.field.field...
497 std::size_t n = 1;
498 while (n < x.constList().size())
499 {
500 if (x.constList()[n].nodeType() == NodeType::GetField)
501 {
502 _compile(x.constList()[n], proc_page);
503 n++;
504 }
505 else
506 break;
507 }
508 std::size_t proc_page_len = m_temp_pages.back().size();
509
510 // we know that operators take only 1 instruction, so if there are more
511 // it's a builtin/function
512 if (proc_page_len > 1)
513 {
514 // push arguments on current page
515 for (auto exp = x.constList().begin() + n, exp_end = x.constList().end(); exp != exp_end; ++exp)
516 _compile(*exp, p);
517 // push proc from temp page
518 for (auto&& inst : m_temp_pages.back())
519 page(p).push_back(inst);
520 m_temp_pages.pop_back();
521
522 // call the procedure
523 page(p).push_back(Instruction::CALL);
524 // number of arguments
525 std::size_t args_count = 0;
526 for (auto it = x.constList().begin() + 1, it_end = x.constList().end(); it != it_end; ++it)
527 {
528 if (it->nodeType() != NodeType::GetField &&
529 it->nodeType() != NodeType::Capture)
530 args_count++;
531 }
532 pushNumber(static_cast<uint16_t>(args_count), &page(p));
533 }
534 else // operator
535 {
536 // retrieve operator
537 auto op_inst = m_temp_pages.back()[0];
538 m_temp_pages.pop_back();
539
540 // push arguments on current page
541 std::size_t exp_count = 0;
542 for (std::size_t index = n, size = x.constList().size(); index < size; ++index)
543 {
544 _compile(x.constList()[index], p);
545
546 if ((index + 1 < size &&
547 x.constList()[index + 1].nodeType() != NodeType::GetField &&
548 x.constList()[index + 1].nodeType() != NodeType::Capture) ||
549 index + 1 == size)
550 exp_count++;
551
552 // in order to be able to handle things like (op A B C D...)
553 // which should be transformed into A B op C op D op...
554 if (exp_count >= 2)
555 page(p).push_back(op_inst);
556 }
557
558 if (exp_count == 1)
559 page(p).push_back(op_inst);
560
561 // need to check we didn't push the (op A B C D...) things for operators not supporting it
562 if (exp_count > 2)
563 {
564 switch (op_inst)
565 {
566 // authorized instructions
567 case Instruction::ADD:
568 case Instruction::SUB:
569 case Instruction::DIV:
570 case Instruction::MUL:
571 case Instruction::MOD:
572 case Instruction::AND_:
573 case Instruction::OR_:
574 break;
575
576 default:
577 throwCompilerError("can not create a chained expression (of length " + std::to_string(exp_count) +
578 ") for operator `" + Builtins::operators[static_cast<std::size_t>(op_inst - Instruction::FIRST_OPERATOR)] +
579 "'. You most likely forgot a `)'.",
580 x);
581 }
582 }
583 }
