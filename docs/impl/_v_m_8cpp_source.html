23 void VM::init() noexcept
24 {
25 // initialize the stack

28
29 m_sp = 0;

45 createNewScope();

53 if (it != m_state->m_symbols.end())
54 (*m_locals[0]).push_back(static_cast<uint16_t>(std::distance(m_state->m_symbols.begin(), it)), name_val.second);

58 Value& VM::operator[](const std::string& name) noexcept
59 {

61
62 // find id of object

78 void VM::loadPlugin(uint16_t id)
79 {
80 namespace fs = std::filesystem;
81
82 const std::string file = m_state->m_constants[id].stringRef().toString();
83
84 std::string path = file;
85 // bytecode loaded from file
86 if (m_state->m_filename != ARK_NO_NAME_FILE)
87 path = (fs::path(m_state->m_filename).parent_path() / fs::path(file)).relative_path().string();
88
89 std::string lib_path = (fs::path(m_state->m_libdir) / fs::path(file)).string();
90
91 // if it's already loaded don't do anything
92 if (std::find_if(m_shared_lib_objects.begin(), m_shared_lib_objects.end(), [&, this](const auto& val) {
93 return (val->path() == path || val->path() == lib_path);
94 }) != m_shared_lib_objects.end())
95 return;
96
97 // if it exists alongside the .arkc file
98 if (Utils::fileExists(path))
99 m_shared_lib_objects.emplace_back(std::make_shared<SharedLibrary>(path));
100 // check in lib_path otherwise
101 else if (Utils::fileExists(lib_path))
102 m_shared_lib_objects.emplace_back(std::make_shared<SharedLibrary>(lib_path));
103 else
104 throwVMError("Could not find module '" + file + "'. Searched in\n\t- " + path + "\n\t- " + lib_path);
105
106 // load the mapping from the dynamic library
107 mapping* map;
108 try
109 {
110 map = m_shared_lib_objects.back()->template get<mapping* (*)()>("getFunctionsMapping")();
111 }
112 catch (const std::system_error& e)
113 {
114 throwVMError(
115 "An error occurred while loading module '" + file + "': " + std::string(e.what()) + "\n" +
116 "It is most likely because the versions of the module and the language don't match.");
117 }
118
119 // load the mapping data
120 std::size_t i = 0;
121 while (map[i].name != nullptr)
122 {
123 // put it in the global frame, aka the first one
124 auto it = std::find(m_state->m_symbols.begin(), m_state->m_symbols.end(), std::string(map[i].name));
125 if (it != m_state->m_symbols.end())
126 (*m_locals[0]).push_back(static_cast<uint16_t>(std::distance(m_state->m_symbols.begin(), it)), Value(map[i].value));
127
128 // free memory because we have used it and don't need it anymore
129 // no need to free map[i].value since it's a pointer to a function in the DLL
130 delete[] map[i].name;
131 ++i;
132 }
133
134 // free memory
135 delete[] map;
136 }

174 int VM::safeRun(std::size_t untilFrameCount)
175 {
176 m_until_frame_count = untilFrameCount;
177
178 try
179 {
180 m_running = true;
181 while (m_running && m_fc > m_until_frame_count)
182 {
183 // get current instruction
184 uint8_t inst = m_state->m_pages[m_pp][m_ip];
185
186 // and it's time to du-du-du-du-duel!
187 switch (inst)
188 {
189#pragma region "Instructions"
190
191 case Instruction::LOAD_SYMBOL:
192 {
193 /*
194 Argument: symbol id (two bytes, big endian)
195 Job: Load a symbol from its id onto the stack
196 */
197
198 ++m_ip;
199 m_last_sym_loaded = readNumber();
200
201 if (Value* var = findNearestVariable(m_last_sym_loaded); var != nullptr)
202 // push internal reference, shouldn't break anything so far
203 push(var);
204 else
205 throwVMError("unbound variable: " + m_state->m_symbols[m_last_sym_loaded]);
206
207 COZ_PROGRESS_NAMED("ark vm load_symbol");
208 break;
209 }
210
211 case Instruction::LOAD_CONST:
212 {
213 /*
214 Argument: constant id (two bytes, big endian)
215 Job: Load a constant from its id onto the stack. Should check for a saved environment
216 and push a Closure with the page address + environment instead of the constant
217 */
218
219 ++m_ip;
220 uint16_t id = readNumber();
221
222 if (m_saved_scope && m_state->m_constants[id].valueType() == ValueType::PageAddr)
223 {
224 push(Value(Closure(m_saved_scope.value(), m_state->m_constants[id].pageAddr())));
225 m_saved_scope.reset();
226 }
227 else
228 {
229 // push internal ref
230 push(&(m_state->m_constants[id]));
231 }
232
233 COZ_PROGRESS_NAMED("ark vm load_const");
234 break;
235 }
236
237 case Instruction::POP_JUMP_IF_TRUE:
238 {
239 /*
240 Argument: absolute address to jump to (two bytes, big endian)
241 Job: Jump to the provided address if the last value on the stack was equal to true.
242 Remove the value from the stack no matter what it is
243 */
244
245 ++m_ip;
246 uint16_t id = readNumber();
247
248 if (*popAndResolveAsPtr() == Builtins::trueSym)
249 m_ip = static_cast<int16_t>(id) - 1; // because we are doing a ++m_ip right after this
250 break;
251 }
252
253 case Instruction::STORE:
254 {
255 /*
256 Argument: symbol id (two bytes, big endian)
257 Job: Take the value on top of the stack and put it inside a variable named following
258 the symbol id (cf symbols table), in the nearest scope. Raise an error if it
259 couldn't find a scope where the variable exists
260 */
261
262 ++m_ip;
263 uint16_t id = readNumber();
264
265 if (Value* var = findNearestVariable(id); var != nullptr)
266 {
267 if (var->isConst())
268 throwVMError("can not modify a constant: " + m_state->m_symbols[id]);
269
270 *var = *popAndResolveAsPtr();
271 var->setConst(false);
272 break;
273 }
274
275 COZ_PROGRESS_NAMED("ark vm store");
276
277 throwVMError("unbound variable " + m_state->m_symbols[id] + ", can not change its value");
278 break;
279 }
280
281 case Instruction::LET:
282 {
283 /*
284 Argument: symbol id (two bytes, big endian)
285 Job: Take the value on top of the stack and create a constant in the current scope, named
286 following the given symbol id (cf symbols table)
287 */
288
289 ++m_ip;
290 uint16_t id = readNumber();
291
292 // check if we are redefining a variable
293 if (auto val = (*m_locals.back())[id]; val != nullptr)
294 throwVMError("can not use 'let' to redefine the variable " + m_state->m_symbols[id]);
295
296 Value val = *popAndResolveAsPtr();
297 val.setConst(true);
298 (*m_locals.back()).push_back(id, val);
299
300 COZ_PROGRESS("ark vm let");
301 break;
302 }
303
304 case Instruction::POP_JUMP_IF_FALSE:
305 {
306 /*
307 Argument: absolute address to jump to (two bytes, big endian)
308 Job: Jump to the provided address if the last value on the stack was equal to false. Remove
309 the value from the stack no matter what it is
310 */
311
312 ++m_ip;
313 uint16_t id = readNumber();
314
315 if (*popAndResolveAsPtr() == Builtins::falseSym)
316 m_ip = static_cast<int16_t>(id) - 1; // because we are doing a ++m_ip right after this
317 break;
318 }
319
320 case Instruction::JUMP:
321 {
322 /*
323 Argument: absolute address to jump to (two byte, big endian)
324 Job: Jump to the provided address
325 */
326
327 ++m_ip;
328 uint16_t id = readNumber();
329
330 m_ip = static_cast<int16_t>(id) - 1; // because we are doing a ++m_ip right after this
331 break;
332 }
333
334 case Instruction::RET:
335 {
336 /*
337 Argument: none
338 Job: If in a code segment other than the main one, quit it, and push the value on top of
339 the stack to the new stack ; should as well delete the current environment.
340 */
341
342 Value ip_or_val = *popAndResolveAsPtr();
343 // no return value on the stack
344 if (ip_or_val.valueType() == ValueType::InstPtr)
345 {
346 m_ip = ip_or_val.pageAddr();
347 // we always push PP then IP, thus the next value
348 // MUST be the page pointer
349 m_pp = pop()->pageAddr();
350
351 returnFromFuncCall();
352 push(Builtins::nil);
353 }
354 // value on the stack
355 else
356 {
357 Value* ip;
358 do
359 {
360 ip = popAndResolveAsPtr();
361 } while (ip->valueType() != ValueType::InstPtr);
362
363 m_ip = ip->pageAddr();
364 m_pp = pop()->pageAddr();
365
366 returnFromFuncCall();
367 push(std::move(ip_or_val));
368 }
369
370 COZ_PROGRESS_NAMED("ark vm ret");
371 break;
372 }
373
374 case Instruction::HALT:
375 m_running = false;
376 break;
377
378 case Instruction::CALL:
379 call();
380 break;
381
382 case Instruction::CAPTURE:
383 {
384 /*
385 Argument: symbol id (two bytes, big endian)
386 Job: Used to tell the Virtual Machine to capture the variable from the current environment.
387 Main goal is to be able to handle closures, which need to save the environment in which
388 they were created
389 */
390
391 ++m_ip;
392 uint16_t id = readNumber();
393
394 if (!m_saved_scope)
395 m_saved_scope = std::make_shared<Scope>();
396 // if it's a captured variable, it can not be nullptr
397 Value* ptr = (*m_locals.back())[id];
398 ptr = ptr->valueType() == ValueType::Reference ? ptr->reference() : ptr;
399 (*m_saved_scope.value()).push_back(id, *ptr);
400
401 COZ_PROGRESS_NAMED("ark vm capture");
402 break;
403 }
404
405 case Instruction::BUILTIN:
406 {
407 /*
408 Argument: id of builtin (two bytes, big endian)
409 Job: Push the builtin function object on the stack
410 */
411
412 ++m_ip;
413 uint16_t id = readNumber();
414
415 push(Builtins::builtins[id].second);
416
417 COZ_PROGRESS_NAMED("ark vm builtin");
418 break;
419 }
420
421 case Instruction::MUT:
422 {
423 /*
424 Argument: symbol id (two bytes, big endian)
425 Job: Take the value on top of the stack and create a variable in the current scope,
426 named following the given symbol id (cf symbols table)
427 */
428
429 ++m_ip;
430 uint16_t id = readNumber();
431
432 Value val = *popAndResolveAsPtr();
433 val.setConst(false);
434
435 // avoid adding the pair (id, _) multiple times, with different values
436 Value* local = (*m_locals.back())[id];
437 if (local == nullptr)
438 (*m_locals.back()).push_back(id, val);
439 else
440 *local = val;
441
442 COZ_PROGRESS_NAMED("ark vm mut");
443 break;
444 }
445
446 case Instruction::DEL:
447 {
448 /*
449 Argument: symbol id (two bytes, big endian)
450 Job: Remove a variable/constant named following the given symbol id (cf symbols table)
451 */
452
453 ++m_ip;
454 uint16_t id = readNumber();
455
456 if (Value* var = findNearestVariable(id); var != nullptr)
457 {
458 // free usertypes
459 if (var->valueType() == ValueType::User)
460 var->usertypeRef().del();
461 *var = Value();
462 break;
463 }
464
465 COZ_PROGRESS_NAMED("ark vm del");
466
467 throwVMError("unbound variable: " + m_state->m_symbols[id]);
468 break;
469 }
470
471 case Instruction::SAVE_ENV:
472 {
473 /*
474 Argument: none
475 Job: Save the current environment, useful for quoted code
476 */
477 m_saved_scope = m_locals.back();
478
479 COZ_PROGRESS("ark vm save_scope");
480 break;
481 }
482
483 case Instruction::GET_FIELD:
484 {
485 /*
486 Argument: symbol id (two bytes, big endian)
487 Job: Used to read the field named following the given symbol id (cf symbols table) of a `Closure`
488 stored in TS. Pop TS and push the value of field read on the stack
489 */
490
491 ++m_ip;
492 uint16_t id = readNumber();
493
494 Value* var = popAndResolveAsPtr();
495 if (var->valueType() != ValueType::Closure)
496 throwVMError("the variable `" + m_state->m_symbols[m_last_sym_loaded] + "' isn't a closure, can not get the field `" + m_state->m_symbols[id] + "' from it");
497
498 if (Value* field = (*var->refClosure().scope())[id]; field != nullptr)
499 {
500 // check for CALL instruction
501 if (m_ip + 1 < m_state->m_pages[m_pp].size() && m_state->m_pages[m_pp][m_ip + 1] == Instruction::CALL)
502 {
503 m_locals.push_back(var->refClosure().scope());
504 ++m_scope_count_to_delete.back();
505 }
506
507 push(field);
508 break;
509 }
510
511 throwVMError("couldn't find the variable " + m_state->m_symbols[id] + " in the closure enviroment");
512 break;
513 }
514
515 case Instruction::PLUGIN:
516 {
517 /*
518 Argument: constant id (two bytes, big endian)
519 Job: Load a module named following the constant id (cf constants table).
520 Raise an error if it couldn't find the module.
521 */
522
523 ++m_ip;
524 uint16_t id = readNumber();
525
526 loadPlugin(id);
527
528 COZ_PROGRESS("ark vm plugin");
529 break;
530 }
531
532 case Instruction::LIST:
533 {
534 /*
535 Takes at least 0 arguments and push a list on the stack.
536 The content is pushed in reverse order
537 */
538 ++m_ip;
539 uint16_t count = readNumber();
540
541 Value l(ValueType::List);
542 if (count != 0)
543 l.list().reserve(count);
544
545 for (uint16_t i = 0; i < count; ++i)
546 l.push_back(*popAndResolveAsPtr());
547 push(std::move(l));
548
549 COZ_PROGRESS_NAMED("ark vm list");
550 break;
551 }
552
553 case Instruction::APPEND:
554 {
555 ++m_ip;
556 uint16_t count = readNumber();
557
558 Value* list = popAndResolveAsPtr();
559 if (list->valueType() != ValueType::List)
560 throw TypeError("Argument 1 of append should be a List, got " + types_to_str[static_cast<unsigned>(list->valueType())]);
561 const uint16_t size = list->constList().size();
562
563 Value obj = Value(*list);
564 obj.list().reserve(size + count);
565
566 for (uint16_t i = 0; i < count; ++i)
567 obj.push_back(*popAndResolveAsPtr());
568 push(std::move(obj));
569
570 COZ_PROGRESS_NAMED("ark vm append");
571 break;
572 }
573
574 case Instruction::CONCAT:
575 {
576 ++m_ip;
577 uint16_t count = readNumber();
578
579 Value* list = popAndResolveAsPtr();
580 if (list->valueType() != ValueType::List)
581 throw TypeError("Argument 1 of concat should be a List, got " + types_to_str[static_cast<unsigned>(list->valueType())]);
582
583 Value obj = Value(*list);
584
585 for (uint16_t i = 0; i < count; ++i)
586 {
587 Value* next = popAndResolveAsPtr();
588 if (next->valueType() != ValueType::List)
589 throw TypeError("Arguments of concat should be Lists, got " + types_to_str[static_cast<unsigned>(next->valueType())]);
590
591 for (auto it = next->list().begin(), end = next->list().end(); it != end; ++it)
592 obj.push_back(*it);
593 }
594 push(std::move(obj));
595
596 COZ_PROGRESS_NAMED("ark vm concat");
597 break;
598 }
599
600 case Instruction::APPEND_IN_PLACE:
601 {
602 ++m_ip;
603 uint16_t count = readNumber();
604
605 Value* list = popAndResolveAsPtr();
606
607 if (list->isConst())
608 throwVMError("can not modify a constant list using `append!'");
609 if (list->valueType() != ValueType::List)
610 throw TypeError("Argument 1 of append! should be a List, got " + types_to_str[static_cast<unsigned>(list->valueType())]);
611
612 for (uint16_t i = 0; i < count; ++i)
613 list->push_back(*popAndResolveAsPtr());

615 push(Nil);
616
617 COZ_PROGRESS_NAMED("ark vm append!");
618 break;

625

627
628 if (list->isConst())

649 case Instruction::POP_LIST:
650 {
651 Value list = *popAndResolveAsPtr();
652 Value number = *popAndResolveAsPtr();
653
654 if (list.valueType() != ValueType::List)
655 throw TypeError("Argument 1 of pop should be a List, got " + types_to_str[static_cast<unsigned>(list.valueType())]);
656 if (number.valueType() != ValueType::Number)

661 if (idx >= list.list().size())
662 throw std::runtime_error("pop: index out of range");
663
664 list.list().erase(list.list().begin() + idx);
665 push(list);

694 case Instruction::ADD:

784 }

786 case Instruction::NEQ:
787 {
788 Value *b = popAndResolveAsPtr(), *a = popAndResolveAsPtr();
789
790 push((*a != *b) ? Builtins::trueSym : Builtins::falseSym);
791 break;
792 }
793
794 case Instruction::EQ:
795 {
796 Value *b = popAndResolveAsPtr(), *a = popAndResolveAsPtr();
797
798 push((*a == *b) ? Builtins::trueSym : Builtins::falseSym);
799 break;
800 }
801
802 case Instruction::LEN:
803 {
804 Value* a = popAndResolveAsPtr();
805
806 if (a->valueType() == ValueType::List)
807 push(Value(static_cast<int>(a->constList().size())));
808 else if (a->valueType() == ValueType::String)
809 push(Value(static_cast<int>(a->string().size())));
810 else
811 throw TypeError("Argument of len must be a List or a String");
812 break;
813 }
814
815 case Instruction::EMPTY:
816 {
817 Value* a = popAndResolveAsPtr();
818
819 if (a->valueType() == ValueType::List)
820 push((a->constList().size() == 0) ? Builtins::trueSym : Builtins::falseSym);
821 else if (a->valueType() == ValueType::String)
822 push((a->string().size() == 0) ? Builtins::trueSym : Builtins::falseSym);
823 else
824 throw TypeError("Argument of empty? must be a List or a String");
825
826 break;
827 }
828
829 case Instruction::TAIL:
830 {
831 Value* a = popAndResolveAsPtr();
832
833 if (a->valueType() == ValueType::List)
834 {
835 if (a->constList().size() < 2)
836 {
837 push(Value(ValueType::List));
838 break;
839 }
840
841 std::vector<Value> tmp(a->constList().size() - 1);
842 for (std::size_t i = 1, end = a->constList().size(); i < end; ++i)
843 tmp[i - 1] = a->constList()[i];
844 push(Value(std::move(tmp)));
845 }
846 else if (a->valueType() == ValueType::String)
847 {
848 if (a->string().size() < 2)
849 {
850 push(Value(ValueType::String));
851 break;
852 }
853
854 Value b = *a;
855 b.stringRef().erase_front(0);
856 push(std::move(b));
857 }
858 else
859 throw TypeError("Argument of tail must be a List or a String");
860
861 break;
862 }
863
864 case Instruction::HEAD:
865 {
866 Value* a = popAndResolveAsPtr();
867
868 if (a->valueType() == ValueType::List)
869 {
870 if (a->constList().size() == 0)
871 {
872 push(Builtins::nil);
873 break;
874 }
875
876 Value b = a->constList()[0];
877 push(b);
878 }
879 else if (a->valueType() == ValueType::String)
880 {
881 if (a->string().size() == 0)
882 {
883 push(Value(ValueType::String));
884 break;
885 }
886
887 push(Value(std::string(1, a->stringRef()[0])));
888 }
889 else
890 throw TypeError("Argument of head must be a List or a String");
891
892 break;
893 }
894
895 case Instruction::ISNIL:
896 {
897 Value* a = popAndResolveAsPtr();
898 push((*a == Builtins::nil) ? Builtins::trueSym : Builtins::falseSym);
899 break;
900 }
901
902 case Instruction::ASSERT:
903 {
904 Value *b = popAndResolveAsPtr(), *a = popAndResolveAsPtr();
905
906 if (*a == Builtins::falseSym)
907 {
908 if (b->valueType() != ValueType::String)
909 throw TypeError("Second argument of assert must be a String");
910
911 throw AssertionFailed(b->stringRef().toString());
912 }
913 break;
914 }
915
916 case Instruction::TO_NUM:
917 {
918 Value* a = popAndResolveAsPtr();
919
920 if (a->valueType() != ValueType::String)
921 throw TypeError("Argument of toNumber must be a String");
922
923 double val;
924 if (Utils::isDouble(a->string().c_str(), &val))
925 push(Value(val));
926 else
927 push(Builtins::nil);
928 break;
929 }
930
931 case Instruction::TO_STR:
932 {
933 std::stringstream ss;
934 Value* a = popAndResolveAsPtr();
935 ss << (*a);
936 push(Value(ss.str()));
937 break;
938 }
939
940 case Instruction::AT:
941 {
942 Value* b = popAndResolveAsPtr();
943 Value a = *popAndResolveAsPtr(); // be careful, it's not a pointer
944
945 if (b->valueType() != ValueType::Number)
946 throw TypeError("Argument 2 of @ should be a Number");
947
948 long idx = static_cast<long>(b->number());
949
950 if (a.valueType() == ValueType::List)
951 push(a.list()[idx < 0 ? a.list().size() + idx : idx]);
952 else if (a.valueType() == ValueType::String)
953 push(Value(std::string(1, a.string()[idx < 0 ? a.string().size() + idx : idx])));
954 else
955 throw TypeError("Argument 1 of @ should be a List or a String");
956 break;
957 }
958
959 case Instruction::AND_:
960 {
961 Value *a = popAndResolveAsPtr(), *b = popAndResolveAsPtr();
962
963 push((*a == Builtins::trueSym && *b == Builtins::trueSym) ? Builtins::trueSym : Builtins::falseSym);
964 break;
965 }
966
967 case Instruction::OR_:
968 {
969 Value *a = popAndResolveAsPtr(), *b = popAndResolveAsPtr();
970
971 push((*b == Builtins::trueSym || *a == Builtins::trueSym) ? Builtins::trueSym : Builtins::falseSym);
972 break;
973 }
974
975 case Instruction::MOD:
976 {
977 Value *b = popAndResolveAsPtr(), *a = popAndResolveAsPtr();
978
979 if (a->valueType() != ValueType::Number)
980 throw TypeError("Arguments of mod should be Numbers");
981 if (b->valueType() != ValueType::Number)
982 throw TypeError("Arguments of mod should be Numbers");
983
984 push(Value(std::fmod(a->number(), b->number())));
985 break;
986 }
987
988 case Instruction::TYPE:
989 {
990 Value* a = popAndResolveAsPtr();
991
992 push(Value(types_to_str[static_cast<unsigned>(a->valueType())]));
993 break;
994 }
995
996 case Instruction::HASFIELD:
997 {
998 Value *field = popAndResolveAsPtr(), *closure = popAndResolveAsPtr();
999
1000 if (closure->valueType() != ValueType::Closure)
1001 throw TypeError("Argument no 1 of hasField should be a Closure");
1002 if (field->valueType() != ValueType::String)
1003 throw TypeError("Argument no 2 of hasField should be a String");
1004
1005 auto it = std::find(m_state->m_symbols.begin(), m_state->m_symbols.end(), field->stringRef().toString());
1006 if (it == m_state->m_symbols.end())
1007 {
1008 push(Builtins::falseSym);
1009 break;
1010 }
1011
1012 uint16_t id = static_cast<uint16_t>(std::distance(m_state->m_symbols.begin(), it));
1013 push((*closure->refClosure().refScope())[id] != nullptr ? Builtins::trueSym : Builtins::falseSym);
1014
1015 break;
1016 }
1017
1018 case Instruction::NOT:
1019 {
1020 Value* a = popAndResolveAsPtr();
1021
1022 push(!(*a) ? Builtins::trueSym : Builtins::falseSym);
1023 break;
1024 }
1025
1026#pragma endregion
1027
1028 default:
1029 throwVMError("unknown instruction: " + std::to_string(static_cast<std::size_t>(inst)));
1030 break;
1031 }
1032
1033 // move forward
1034 ++m_ip;
1035 }
1036 }
1037 catch (const std::exception& e)
1038 {
1039 std::printf("%s\n", e.what());
1040 backtrace();
1041 m_exit_code = 1;
1042 }
1043 catch (...)
1044 {
1045 std::printf("Unknown error\n");
1046 backtrace();
1047 m_exit_code = 1;
1048 }
1049
1050 return m_exit_code;
1051 }

1102 ip = popAndResolveAsPtr();

1104
1105 m_ip = ip->pageAddr();

1118 std::printf("...\n");

1120 }
1121 }
1122
1123 // display variables values in the current scope
1124 std::printf("\nCurrent scope variables values:\n");
1125 for (std::size_t i = 0, size = old_scope.size(); i < size; ++i)
1126 std::cerr << termcolor::cyan << m_state->m_symbols[old_scope.m_data[i].first] << termcolor::reset

1128
1129 while (m_fc != 1)

1131 Value* tmp = pop();

1133 --m_fc;
1134 else if (tmp->valueType() == ValueType::User)
