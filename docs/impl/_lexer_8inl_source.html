4inline TokenType Lexer::guessType(const std::string& value) noexcept
5{
6 if (value.empty())
7 return TokenType::Mismatch;
8
9 // assuming we already detected ()[]{}, strings, shorthands and comments
10 if (Utils::isDouble(value)) // works on (\+|-)?[[:digit:]]+(\.[[:digit:]]+)?([e|E](\+|-)?[[:digit]]+)?
11 return TokenType::Number;
12 else if (isOperator(value))
13 return TokenType::Operator;
14 else if (isKeyword(value))
15 return TokenType::Keyword;
16 else if (value[0] == '&' && value.size() > 1 && isIdentifier(value))
17 return TokenType::Capture;
18 else if (value.size() > 3 && value[0] == value[1] && value[1] == value[2] && value[2] == '.')
19 return TokenType::Spread;
20 else if (value[0] == '.' && value.size() > 1 && isIdentifier(value))
21 return TokenType::GetField;
22 // otherwise, identifier if it starts with [a-zA-Z_]
23 else if (isIdentifier(value))
24 return TokenType::Identifier;
25 return TokenType::Mismatch;
26}

44{
45 switch (sequence[0])
