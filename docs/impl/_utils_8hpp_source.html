35 inline void stringReplaceAll(std::string& source, const std::string& from, const std::string& to)
36 {
37 std::string newString;
38 newString.reserve(source.length()); // avoids a few memory allocations
39
40 std::string::size_type lastPos = 0;
41 std::string::size_type findPos;
42
43 while (std::string::npos != (findPos = source.find(from, lastPos)))
44 {
45 newString.append(source, lastPos, findPos - lastPos);
46 newString += to;
47 lastPos = findPos + from.length();
48 }
49
50 // Care for the rest after last occurrence
51 newString += source.substr(lastPos);
52
53 source.swap(newString);
54 }
