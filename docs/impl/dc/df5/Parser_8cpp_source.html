21 void Parser::feed(const std::string& code, const std::string& filename)
22 {
23 // not the default value
24 if (filename != ARK_NO_NAME_FILE)
25 {
26 m_file = Utils::canonicalRelPath(filename);
27 if (m_debug >= 2)
28 std::cout << "New parser: " << m_file << '\n';
29 m_parent_include.push_back(m_file);
30 }
31
32 m_code = code;
33
34 m_lexer.feed(code);
35 // apply syntactic sugar
36 std::vector<Token>& t = m_lexer.tokens();
37 if (t.empty())
38 throw ParseError("empty file");
39 sugar(t);
40
41 // create program
42 std::list<Token> tokens(t.begin(), t.end());
43 m_last_token = tokens.front();
44
45 // accept every nodes in the file
46 m_ast = Node(NodeType::List);
47 m_ast.setFilename(m_file);
48 m_ast.list().emplace_back(Keyword::Begin);
49 while (!tokens.empty())
50 m_ast.list().push_back(parse(tokens));
51 // include files if needed
52 checkForInclude(m_ast, m_ast);
53
54 if (m_debug >= 3)
55 std::cout << "(Parser) AST\n"
56 << m_ast << "\n\n";
57 }

69 void Parser::sugar(std::vector<Token>& tokens) noexcept
70 {
71 std::size_t i = 0;
72 while (true)
73 {
74 std::size_t line = tokens[i].line;
75 std::size_t col = tokens[i].col;
76
77 if (tokens[i].token == "{")
78 {
79 tokens[i] = Token(TokenType::Grouping, "(", line, col);
80 // handle macros
81 if (i > 0 && tokens[i - 1].token != "!")
82 tokens.insert(tokens.begin() + i + 1, Token(TokenType::Keyword, "begin", line, col));
83 else if (i == 0)
84 tokens.insert(tokens.begin() + i + 1, Token(TokenType::Keyword, "begin", line, col));
85 }
86 else if (tokens[i].token == "}" || tokens[i].token == "]")
87 tokens[i] = Token(TokenType::Grouping, ")", line, col);
88 else if (tokens[i].token == "[")
89 {
90 tokens[i] = Token(TokenType::Grouping, "(", line, col);
91 tokens.insert(tokens.begin() + i + 1, Token(TokenType::Identifier, "list", line, col));
92 }
93
94 ++i;
95
96 if (i == tokens.size())
97 break;
98 }
99 }

101 // sugar() was called before, so it's safe to assume we only have ( and )
102 Node Parser::parse(std::list<Token>& tokens, bool authorize_capture, bool authorize_field_read, bool in_macro)
103 {
104 using namespace std::string_literals;
105
106 Token token = nextToken(tokens);
107
108 bool previous_token_was_lparen = false;
109
110 // parse block
111 if (token.token == "(")
112 {
113 previous_token_was_lparen = true;
114 // create a list node to host the block
115 Node block = make_node_list(token.line, token.col, m_file);
116
117 // handle sub-blocks
118 if (tokens.front().token == "(")
119 {
120 block.push_back(parse(tokens, false, false, in_macro));
121 previous_token_was_lparen = false;
122 }
123
124 // take next token, we don't want to play with a "("
125 token = nextToken(tokens);
126
127 // return an empty block
128 if (token.token == ")")
129 return block;
130
131 // check for unexpected keywords between expressions
132 if ((token.type == TokenType::Operator ||
133 token.type == TokenType::Identifier ||
134 token.type == TokenType::Number ||
135 token.type == TokenType::String) &&
136 tokens.front().type == TokenType::Keyword)
137 throwParseError("Unexpected keyword `" + tokens.front().token + "' in the middle of an expression", tokens.front());
138
139 // loop until we reach the end of the block
140 do
141 {
142 Node atomized = atom(token);
143 checkForInvalidTokens(atomized, token, previous_token_was_lparen, authorize_capture, authorize_field_read);
144 block.push_back(atomized);
145
146 expect(!tokens.empty(), "expected more tokens after `" + token.token + "'", m_last_token);
147 m_last_token = tokens.front();
148
149 if (token.type == TokenType::Keyword)
150 {
151 void (Parser::*fun_ptr)(Node&, Token&, std::list<Token>&, bool, bool, bool) = nullptr;
152 if (token.token == "if")
153 fun_ptr = &Parser::parseIf;
154 else if (token.token == "let" || token.token == "mut")
155 fun_ptr = &Parser::parseLetMut;
156 else if (token.token == "set")
157 fun_ptr = &Parser::parseSet;
158 else if (token.token == "fun")
159 fun_ptr = &Parser::parseFun;
160 else if (token.token == "while")
161 fun_ptr = &Parser::parseWhile;
162 else if (token.token == "begin")
163 fun_ptr = &Parser::parseBegin;
164 else if (token.token == "import")
165 fun_ptr = &Parser::parseImport;
166 else if (token.token == "quote")
167 fun_ptr = &Parser::parseQuote;
168 else if (token.token == "del")
169 fun_ptr = &Parser::parseDel;
170
171 if (fun_ptr != nullptr)
172 (this->*fun_ptr)(block, token, tokens, authorize_capture, authorize_field_read, in_macro);
173 else
174 throwParseError("unimplemented keyword `" + token.token + "'. If you see this error please report it on GitHub.", token);
175 }
176 else if (token.type == TokenType::Identifier || token.type == TokenType::Operator ||
177 (token.type == TokenType::Capture && authorize_capture) ||
178 (token.type == TokenType::GetField && authorize_field_read) ||
179 (token.type == TokenType::Spread && in_macro))
180 {
181 while (tokens.front().token != ")")
182 block.push_back(parse(tokens, /* authorize_capture */ false, /* authorize_field_read */ true, in_macro));
183 }
184 } while (tokens.front().token != ")");
185
186 // pop the ")"
187 tokens.pop_front();
188 return block;
189 }
190 else if (token.type == TokenType::Shorthand)
191 return parseShorthand(token, tokens, authorize_capture, authorize_field_read, in_macro);
192 // error, we shouldn't have grouping token here
193 else if (token.type == TokenType::Grouping)
194 throwParseError("Found a lonely `" + token.token + "', you most likely have too much parenthesis.", token);
195 else if ((token.type == TokenType::Operator || token.type == TokenType::Identifier) &&
196 std::find(internal::operators.begin(), internal::operators.end(), token.token) != internal::operators.end())
197 throwParseError("Found a free flying operator, which isn't authorized. Operators should always immediatly follow a `('.", token);
198 else if ((token.type == TokenType::Number ||
199 token.type == TokenType::String) &&
200 tokens.front().type == TokenType::Keyword)
201 throwParseError("Unexpected keyword `" + tokens.front().token + "' in the middle of an expression", tokens.front());
202 else if (token.type == TokenType::Keyword &&
203 !previous_token_was_lparen)
204 throwParseError("Unexpected keyword `" + token.token + "' in the middle of an expression", token);
205 return atom(token);
206 }

208 void Parser::parseIf(Node& block, Token& token [[maybe_unused]], std::list<Token>& tokens, bool authorize_capture [[maybe_unused]], bool authorize_field_read [[maybe_unused]], bool in_macro)
209 {
210 auto temp = tokens.front();
211 // parse condition
212 if (temp.type == TokenType::Grouping)
213 block.push_back(parse(tokens, false, false, in_macro));
214 else if (temp.type == TokenType::Identifier || temp.type == TokenType::Number ||
215 temp.type == TokenType::String || (in_macro && temp.type == TokenType::Spread))
216 block.push_back(atom(nextToken(tokens)));
217 else
218 throwParseError("found invalid token after keyword `if', expected function call, value or Identifier", temp);
219 // parse 'then'
220 expect(!tokens.empty() && tokens.front().token != ")", "expected a statement after the condition", temp);
221 block.push_back(parse(tokens, false, false, in_macro));
222 // parse 'else', if there is one
223 if (tokens.front().token != ")")
224 {
225 block.push_back(parse(tokens, false, false, in_macro));
226 // error handling if the if is ill-formed
227 expect(tokens.front().token == ")", "if block is ill-formed, got more than the 3 required arguments (condition, then, else)", m_last_token);
228 }
229 }

231 void Parser::parseLetMut(Node& block, Token& token, std::list<Token>& tokens, bool authorize_capture [[maybe_unused]], bool authorize_field_read [[maybe_unused]], bool in_macro)
232 {
233 auto temp = tokens.front();
234 // parse identifier
235 if (temp.type == TokenType::Identifier)
236 block.push_back(atom(nextToken(tokens)));
237 else if (in_macro)
238 block.push_back(parse(tokens, false, false, in_macro));
239 else
240 throwParseError(std::string("missing identifier to define a ") + (token.token == "let" ? "constant" : "variable") + ", after keyword `" + token.token + "'", temp);
241 expect(!tokens.empty() && tokens.front().token != ")", "expected a value after the identifier", temp);
242 // value
243 while (tokens.front().token != ")")
244 block.push_back(parse(tokens, /* authorize_capture */ false, /* authorize_field_read */ true, in_macro));
245
246 // the block size can exceed 3 only if we have a serie of getfields
247 expect(
248 block.list().size() <= 3 || std::all_of(block.list().begin() + 3, block.list().end(), [](const Node& n) -> bool {
249 return n.nodeType() == NodeType::GetField;
250 }),
251 "too many arguments given to keyword `" + token.token + "', got " + std::to_string(block.list().size() - 1) + ", expected at most 3", m_last_token);
252 }

254 void Parser::parseSet(Node& block, Token& token, std::list<Token>& tokens, bool authorize_capture [[maybe_unused]], bool authorize_field_read [[maybe_unused]], bool in_macro)
255 {
256 auto temp = tokens.front();
257 // parse identifier
258 if (temp.type == TokenType::Identifier)
259 block.push_back(atom(nextToken(tokens)));
260 else if (in_macro)
261 block.push_back(parse(tokens, false, false, in_macro));
262 else
263 throwParseError("missing identifier to assign a value to, after keyword `set'", temp);
264 expect(!tokens.empty() && tokens.front().token != ")", "expected a value after the identifier", temp);
265 // set can not accept a.b...c as an identifier
266 if (tokens.front().type == TokenType::GetField)
267 throwParseError("found invalid token after keyword `set', expected an identifier, got a closure field reading expression", tokens.front());
268 // value
269 while (tokens.front().token != ")")
270 block.push_back(parse(tokens, /* authorize_capture */ false, /* authorize_field_read */ true, in_macro));
271
272 // the block size can exceed 3 only if we have a serie of getfields
273 expect(
274 block.list().size() <= 3 || std::all_of(block.list().begin() + 3, block.list().end(), [](const Node& n) -> bool {
275 return n.nodeType() == NodeType::GetField;
276 }),
277 "too many arguments given to keyword `" + token.token + "', got " + std::to_string(block.list().size() - 1) + ", expected at most 3", m_last_token);
278 }

280 void Parser::parseFun(Node& block, Token& token, std::list<Token>& tokens, bool authorize_capture [[maybe_unused]], bool authorize_field_read [[maybe_unused]], bool in_macro)
281 {
282 // parse arguments
283 if (tokens.front().type == TokenType::Grouping || in_macro)
284 block.push_back(parse(tokens, /* authorize_capture */ true, false, in_macro));
285 else
286 throwParseError("found invalid token after keyword `fun', expected a block to define the argument list of the function\nThe block can be empty if it doesn't have arguments: `()'", tokens.front());
287 // parse body
288 if (tokens.front().type == TokenType::Grouping || in_macro)
289 block.push_back(parse(tokens, false, false, in_macro));
290 else
291 throwParseError("the body of a function must be a block, even an empty one `()'", tokens.front());
292 expect(block.list().size() == 3, "got too many arguments after keyword `" + token.token + "', expected an argument list and a body", m_last_token);
293 }

295 void Parser::parseWhile(Node& block, Token& token, std::list<Token>& tokens, bool authorize_capture [[maybe_unused]], bool authorize_field_read [[maybe_unused]], bool in_macro)
296 {
297 auto temp = tokens.front();
298 // parse condition
299 if (temp.type == TokenType::Grouping)
300 block.push_back(parse(tokens, false, false, in_macro));
301 else if (temp.type == TokenType::Identifier || temp.type == TokenType::Number ||
302 temp.type == TokenType::String)
303 block.push_back(atom(nextToken(tokens)));
304 else
305 throwParseError("found invalid token after keyword `while', expected function call, value or Identifier", temp);
306 expect(!tokens.empty() && tokens.front().token != ")", "expected a body after the condition", temp);
307 // parse 'do'
308 block.push_back(parse(tokens, false, false, in_macro));
309 expect(block.list().size() == 3, "got too many arguments after keyword `" + token.token + "', expected a condition and a body", temp);
310 }

349 Node Parser::parseShorthand(Token& token, std::list<Token>& tokens, bool authorize_capture [[maybe_unused]], bool authorize_field_read [[maybe_unused]], bool in_macro)
350 {
351 if (token.token == "'")
352 {
353 // create a list node to host the block
354 Node block = make_node_list(token.line, token.col, m_file);
355
356 block.push_back(make_node(Keyword::Quote, token.line, token.col, m_file));
357 block.push_back(parse(tokens, false, false, in_macro));
358 return block;
359 }
360 else if (token.token == "!")
361 {
362 if (m_debug >= 2)
363 std::cout << "Found a macro at " << token.line << ':' << token.col << " in " << m_file << '\n';
364
365 // macros
366 Node block = make_node(NodeType::Macro, token.line, token.col, m_file);
367
368 Node parsed = parse(tokens, /* authorize_capture */ false, /* authorize_field_read */ false, /* in_macro */ true);
369 if (parsed.nodeType() != NodeType::List || parsed.list().size() < 2 || parsed.list().size() > 4)
370 throwParseError("Macros can only defined using the !{ name value } or !{ name (args) value } syntax", token);
371
372 // append the nodes of the parsed node to the current macro node
373 for (std::size_t i = 0, end = parsed.list().size(); i < end; ++i)
374 block.push_back(parsed.list()[i]);
375 return block;
376 }
377
378 throwParseError("unknown shorthand", token);
379
380 return Node();
381 }

383 void Parser::checkForInvalidTokens(Node& atomized, Token& token, bool previous_token_was_lparen, bool authorize_capture, bool authorize_field_read)
384 {
385 if ((atomized.nodeType() == NodeType::String || atomized.nodeType() == NodeType::Number ||
386 atomized.nodeType() == NodeType::List) &&
387 previous_token_was_lparen)
388 {
389 std::stringstream ss;
390 ss << "found invalid token after `(', expected Keyword, Identifier";
391 if (!authorize_capture && !authorize_field_read)
392 ss << " or Operator";
393 else
394 {
395 ss << ", Operator";
396 if (authorize_capture && !authorize_field_read)
397 ss << " or Capture";
398 else if (!authorize_capture && authorize_field_read)
399 ss << " or GetField";
400 else
401 ss << ", Capture or GetField";
402 }
403 throwParseError(ss.str(), token);
404 }
405 }

407 Token Parser::nextToken(std::list<Token>& tokens)
408 {
409 expect(!tokens.empty(), "no more token to consume", m_last_token);
410 m_last_token = tokens.front();
411
412 const Token out = std::move(tokens.front());
413 tokens.pop_front();
414 return out;
415 }

417 Node Parser::atom(const Token& token)
418 {
419 switch (token.type)
420 {
421 case TokenType::Number:
422 return make_node(std::stod(token.token), token.line, token.col, m_file);
423
424 case TokenType::String:
425 {
426 std::string str = token.token;
427 // remove the " at the beginning and at the end
428 str.erase(0, 1);
429 str.erase(token.token.size() - 2, 1);
430
431 return make_node(str, token.line, token.col, m_file);
432 }
433
434 case TokenType::Keyword:
435 {
436 std::optional<Keyword> kw;
437 if (token.token == "if")
438 kw = Keyword::If;
439 else if (token.token == "set")
440 kw = Keyword::Set;
441 else if (token.token == "let")
442 kw = Keyword::Let;
443 else if (token.token == "mut")
444 kw = Keyword::Mut;
445 else if (token.token == "fun")
446 kw = Keyword::Fun;
447 else if (token.token == "while")
448 kw = Keyword::While;
449 else if (token.token == "begin")
450 kw = Keyword::Begin;
451 else if (token.token == "import")
452 kw = Keyword::Import;
453 else if (token.token == "quote")
454 kw = Keyword::Quote;
455 else if (token.token == "del")
456 kw = Keyword::Del;
457
458 if (kw)
459 return make_node(kw.value(), token.line, token.col, m_file);
460 throwParseError("unknown keyword", token);
461 }
462
463 case TokenType::Capture:
464 case TokenType::GetField:
465 case TokenType::Spread:
466 {
467 Node n = make_node(internal::similarNodetypeFromTokentype(token.type), token.line, token.col, m_file);
468 n.setString(token.type != TokenType::Spread ? token.token : token.token.substr(3));
469 return n;
470 }
471
472 case TokenType::Shorthand:
473 throwParseError("got a shorthand to atomize, and that's not normal. If you see this error please report it on GitHub.", token);
474
475 default:
476 {
477 // assuming it is a TokenType::Identifier, thus a Symbol
478 Node n = make_node(NodeType::Symbol, token.line, token.col, m_file);
479 n.setString(token.token);
480 return n;
481 }
482 }

485 // high cpu cost
486 bool Parser::checkForInclude(Node& n, Node& parent, std::size_t pos)
487 {
488 namespace fs = std::filesystem;
489
490 if (n.nodeType() == NodeType::List)
491 {
492 if (n.constList().size() == 0)
493 return false;
494
495 const Node& first = n.constList()[0];
496
497 if (first.nodeType() == NodeType::Keyword && first.keyword() == Keyword::Import)
498 {
499 if (m_debug >= 2)
500 std::cout << "Import found in file: " << m_file << '\n';
501
502 std::string file;
503 if (n.constList()[1].nodeType() == NodeType::String)
504 file = n.constList()[1].string();
505 else
506 throw TypeError("Arguments of import must be of type String");
507
508 // check if we are not loading a plugin
509 if (fs::path(file).extension().string() == ".ark")
510 {
511 // search for the source file everywhere
512 std::string included_file = seekFile(file);
513
514 // if the file isn't in the include list, then we can include it
515 // this avoids cyclic includes
516 if (std::find(m_parent_include.begin(), m_parent_include.end(), Utils::canonicalRelPath(included_file)) != m_parent_include.end())
517 return true;
518
519 Parser p(m_debug, m_options, m_libenv);
520 // feed the new parser with our parent includes
521 for (auto const& pi : m_parent_include)
522 p.m_parent_include.push_back(pi); // new parser, we can assume that the parent include list is empty
523 p.m_parent_include.push_back(m_file); // add the current file to avoid importing it again
524 p.feed(Utils::readFile(included_file), included_file);
525
526 // update our list of included files
527 for (auto const& inc : p.m_parent_include)
528 {
529 if (std::find(m_parent_include.begin(), m_parent_include.end(), inc) == m_parent_include.end())
530 m_parent_include.push_back(inc);
531 }
532
533 for (std::size_t j = 1, end = p.ast().constList().size(); j < end; ++j)
534 {
535 parent.list().insert(parent.list().begin() + pos + j, p.ast().constList()[j]);
536 }
537
538 return true;
539 }
540 }
541
542 for (std::size_t i = 0; i < n.list().size(); ++i)
543 {
544 if (checkForInclude(n.list()[i], n, i))
545 {
546 n.list().erase(n.list().begin() + i);
547 --i;
548 }
549 }
550 }
551
552 return false;
553 }

555 std::string Parser::seekFile(const std::string& file)
556 {
557 const std::string current_dir = Utils::getDirectoryFromPath(m_file) + "/";
558 const std::string path = (current_dir != "/") ? current_dir + file : file;
559
560 if (m_debug >= 2)
561 {
562 std::cout << "path: " << path << " ; file: " << file << " ; libpath: ";
563 for (auto&& lib : m_libenv)
564 std::cout << lib << ":";
565 std::cout << "\nfilename: " << Utils::getFilenameFromPath(file) << '\n';
566 }
567
568 // search in the current directory
569 if (Utils::fileExists(path))
570 return path;
571
572 // search in all folders in environment path
573 for (auto const& p : m_libenv)
574 {
575 // then search in the standard library directory
576 if (std::string f = p + "/std/" + file; Utils::fileExists(f))
577 return f;
578 // then in the standard library root directory
579 else if (std::string f2 = p + "/" + file; Utils::fileExists(f2))
580 return f2;
581 }
582
583 // fallback, we couldn't find the file
584 throw std::runtime_error("While processing file " + m_file + ", couldn't import " + file + ": file not found");
585 }

593 void Parser::throwParseError(const std::string& message, internal::Token token)
594 {
595 std::stringstream ss;
596 ss << message << "\nGot TokenType::" << internal::tokentype_string[static_cast<unsigned>(token.type)] << "\n";
597
598 if (m_file != ARK_NO_NAME_FILE)
599 ss << "In file " << m_file << "\n";
600 ss << internal::makeTokenBasedErrorCtx(token.token, token.line, token.col, m_code);
601
602 throw ParseError(ss.str());
603 }
