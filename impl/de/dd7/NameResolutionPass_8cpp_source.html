108 void NameResolutionPass::visit(const Node& node)
109 {
110 switch (node.nodeType())
111 {
112 case NodeType::Symbol:
113 addSymbolNode(node);
114 break;
115
116 case NodeType::Field:
117 for (const auto& child : node.constList())
118 addSymbolNode(child);
119 break;
120
121 case NodeType::List:
122 if (!node.constList().empty())
123 {
124 if (node.constList()[0].nodeType() == NodeType::Keyword)
125 visitKeyword(node, node.constList()[0].keyword());
126 else
127 {
128 // function calls
129 // the UpdateRef function calls kind get a special treatment, like let/mut/set,
130 // because we need to check for mutability errors
131 if (node.constList().size() > 1 && node.constList()[0].nodeType() == NodeType::Symbol &&
132 node.constList()[1].nodeType() == NodeType::Symbol)
133 {
134 const auto funcname = node.constList()[0].string();
135 const auto arg = node.constList()[1].string();
136
137 if (std::ranges::find(Language::UpdateRef, funcname) != Language::UpdateRef.end() && m_scope_resolver.isImmutable(arg).value_or(false))
138 throw CodeError(
139 fmt::format("MutabilityError: Can not modify the constant list `{}' using `{}'", arg, funcname),
140 node.filename(),
141 node.constList()[1].line(),
142 node.constList()[1].col(),
143 arg);
144
145 // check that we aren't doing a (append! a a) nor a (concat! a a)
146 if (funcname == Language::AppendInPlace || funcname == Language::ConcatInPlace)
147 {
148 for (std::size_t i = 2, end = node.constList().size(); i < end; ++i)
149 {
150 if (node.constList()[i].nodeType() == NodeType::Symbol && node.constList()[i].string() == arg)
151 throw CodeError(
152 fmt::format("MutabilityError: Can not {} the list `{}' to itself", funcname, arg),
153 node.filename(),
154 node.constList()[1].line(),
155 node.constList()[1].col(),
156 arg);
157 }
158 }
159 }
160
161 for (const auto& child : node.constList())
162 visit(child);
163 }
164 }
165 break;
166
167 default:
168 break;
169 }
170 }

172 void NameResolutionPass::visitKeyword(const Node& node, const Keyword keyword)
173 {
174 switch (keyword)
175 {
176 case Keyword::Set:
177 [[fallthrough]];
178 case Keyword::Let:
179 [[fallthrough]];
180 case Keyword::Mut:
181 // first, visit the value, then register the symbol
182 // this allows us to detect things like (let foo (fun (&foo) ()))
183 if (node.constList().size() > 2)
184 visit(node.constList()[2]);
185 if (node.constList().size() > 1 && node.constList()[1].nodeType() == NodeType::Symbol)
186 {
187 const std::string& name = node.constList()[1].string();
188 if (m_language_symbols.contains(name))
189 throw CodeError(
190 fmt::format("Can not use a reserved identifier ('{}') as a {} name.", name, keyword == Keyword::Let ? "constant" : "variable"),
191 node.filename(),
192 node.constList()[1].line(),
193 node.constList()[1].col(),
194 name);
195
196 if (m_scope_resolver.isInScope(name) && keyword == Keyword::Let)
197 throw CodeError(
198 fmt::format("MutabilityError: Can not use 'let' to redefine variable `{}'", name),
199 node.filename(),
200 node.constList()[1].line(),
201 node.constList()[1].col(),
202 name);
203 else if (keyword == Keyword::Set)
204 {
205 const auto val = node.constList()[2].repr();
206
207 if (const auto mutability = m_scope_resolver.isImmutable(name); m_scope_resolver.isRegistered(name) &&
208 mutability.value_or(false))
209 throw CodeError(
210 fmt::format("MutabilityError: Can not set the constant `{}' to {}", name, val),
211 node.filename(),
212 node.constList()[1].line(),
213 node.constList()[1].col(),
214 name);
215 }
216 else
217 addDefinedSymbol(name, keyword != Keyword::Let);
218 }
219 break;
220
221 case Keyword::Import:
222 if (!node.constList().empty())
223 m_plugin_names.push_back(node.constList()[1].constList().back().string());
224 break;
225
226 case Keyword::Fun:
227 // create a new scope to track variables
228 m_scope_resolver.createNew();
229 if (node.constList()[1].nodeType() == NodeType::List)
230 {
231 for (const auto& child : node.constList()[1].constList())
232 {
233 if (child.nodeType() == NodeType::Capture)
234 {
235 // First, check that the capture is a defined symbol
236 if (!m_defined_symbols.contains(child.string()))
237 {
238 // we didn't find node in the defined symbol list, thus we can't capture node
239 throw CodeError(
240 fmt::format("Can not capture {} because it is referencing an unbound variable.", child.string()),
241 child.filename(),
242 child.line(),
243 child.col(),
244 child.repr());
245 }
246 else if (!m_scope_resolver.isRegistered(child.string()))
247 {
248 throw CodeError(
249 fmt::format("Can not capture {} because it is referencing a variable defined in an unreachable scope.", child.string()),
250 child.filename(),
251 child.line(),
252 child.col(),
253 child.repr());
254 }
255 addDefinedSymbol(child.string(), /* is_mutable= */ true);
256 }
257 else if (child.nodeType() == NodeType::Symbol)
258 addDefinedSymbol(child.string(), /* is_mutable= */ true);
259 }
260 }
261 if (node.constList().size() > 2)
262 visit(node.constList()[2]);
263 m_scope_resolver.removeLocalScope(); // and remove it once the function has been compiled
264 break;
265
266 default:
267 for (const auto& child : node.constList())
268 visit(child);
269 break;
270 }
271 }

299 void NameResolutionPass::checkForUndefinedSymbol() const
300 {
301 for (const auto& sym : m_symbol_nodes)
302 {
303 const auto& str = sym.string();
304 const bool is_plugin = mayBeFromPlugin(str);
305
306 if (!m_defined_symbols.contains(str) && !is_plugin)
307 {
308 std::string message;
309
310 const std::string suggestion = offerSuggestion(str);
311 if (suggestion.empty())
312 message = fmt::format(R"(Unbound variable error "{}" (variable is used but not defined))", str);
313 else
314 message = fmt::format(R"(Unbound variable error "{}" (did you mean "{}"?))", str, suggestion);
315
316 throw CodeError(message, sym.filename(), sym.line(), sym.col(), sym.repr());
317 }
318 }

321 std::string NameResolutionPass::offerSuggestion(const std::string& str) const
322 {
323 std::string suggestion;
324 // our suggestion shouldn't require more than half the string to change
325 std::size_t suggestion_distance = str.size() / 2;
326
327 for (const std::string& symbol : m_defined_symbols)
328 {
329 const std::size_t current_distance = Utils::levenshteinDistance(str, symbol);
330 if (current_distance <= suggestion_distance)
331 {
332 suggestion_distance = current_distance;
333 suggestion = symbol;
334 }
335 }
336
337 return suggestion;
338 }
