166 inline bool operator==(const Value& A, const Value& B) noexcept
167 {
168 // values should have the same type
169 if (A.m_type != B.m_type)
170 return false;
171 // all the types >= Nil are Nil itself, True, False, Undefined
172 if (A.typeNum() >= static_cast<uint8_t>(ValueType::Nil))
173 return true;
174
175 return A.m_value == B.m_value;
176 }
177
178 inline bool operator<(const Value& A, const Value& B) noexcept
179 {
180 if (A.m_type != B.m_type)
181 return (A.typeNum() - B.typeNum()) < 0;
182 return A.m_value < B.m_value;
183 }
184
185 inline bool operator!=(const Value& A, const Value& B) noexcept
186 {
187 return !(A == B);
188 }
189
190 inline bool operator!(const Value& A) noexcept
191 {
192 switch (A.valueType())
193 {
194 case ValueType::List:
195 return A.constList().empty();
196
197 case ValueType::Number:
198 return A.number() == 0.0;
199
200 case ValueType::String:
201 return A.string().empty();
202
203 case ValueType::User:
204 case ValueType::Nil:
205 case ValueType::False:
206 return true;
207
208 case ValueType::True:
209 return false;
210
211 default:
212 return false;
213 }
214 }
