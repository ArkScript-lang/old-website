25 void colorizeLine(const std::string& line, LineColorContextCounts& line_color_context_counts, std::ostream& ss)
26 {
27 // clang-format off

29 termcolor::bright_blue,
30 termcolor::bright_green,
31 termcolor::bright_yellow
32 };
33 // clang-format on
34 constexpr std::size_t pairing_color_size = pairing_color.size();

38 if (isPairableChar(c))

43 {
44 case '(':

46 line_color_context_counts.open_parentheses++;
47 break;
48 case ')':
49 line_color_context_counts.open_parentheses--;
50 pairing_color_index = std::abs(line_color_context_counts.open_parentheses) % pairing_color_size;
51 break;
52 case '[':
53 pairing_color_index = std::abs(line_color_context_counts.open_square_braces) % pairing_color_size;
54 line_color_context_counts.open_square_braces++;
55 break;
56 case ']':
57 line_color_context_counts.open_square_braces--;
58 pairing_color_index = std::abs(line_color_context_counts.open_square_braces) % pairing_color_size;
59 break;
60 case '{':
61 pairing_color_index = std::abs(line_color_context_counts.open_curly_braces) % pairing_color_size;
62 line_color_context_counts.open_curly_braces++;
63 break;
64 case '}':
65 line_color_context_counts.open_curly_braces--;
66 pairing_color_index = std::abs(line_color_context_counts.open_curly_braces) % pairing_color_size;
67 break;
68 default:
69 break;
70 }

73 }

80 {
81 const std::vector<std::string> ctx = Utils::splitString(code, '\n');
82
83 const std::size_t first_line = target_line >= 3 ? target_line - 3 : 0;
84 const std::size_t last_line = (target_line + 3) <= ctx.size() ? target_line + 3 : ctx.size();
85 std::size_t overflow = (col_start + sym_size < ctx[target_line].size()) ? 0 : col_start + sym_size - ctx[target_line].size(); // number of characters that are on more lines below
86 LineColorContextCounts line_color_context_counts;
87
88 for (auto i = first_line; i < last_line; ++i)
89 {
90 os << termcolor::green << std::setw(5) << (i + 1) << termcolor::reset << " | ";
91 colorizeLine(ctx[i], line_color_context_counts, os);
92 os << "\n";
93
94 if (i == target_line || (i > target_line && overflow > 0))
95 {
96 os << " |";
97 // if we have an overflow then we start at the beginning of the line
98 const std::size_t curr_col_start = (overflow == 0) ? col_start : 0;
99 // if we have an overflow, it is used as the end of the line
100 const std::size_t col_end = (i == target_line) ? std::min<std::size_t>(col_start + sym_size, ctx[target_line].size())
101 : std::min<std::size_t>(overflow, ctx[i].size());
102 // update the overflow to avoid going here again if not needed
103 overflow = (overflow > ctx[i].size()) ? overflow - ctx[i].size() : 0;
104
105 // fixing padding when the error is on the first character
106 if (curr_col_start == 0)
107 os << " ";
108
109 // padding of spaces
110 for (std::size_t j = 0; j < curr_col_start; ++j)
111 os << " ";
112
113 // underline the error
114 os << termcolor::red << "^";
115 for (std::size_t j = curr_col_start + 1; j < col_end; ++j)
116 os << "~";
