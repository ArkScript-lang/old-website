62 if (x.nodeType() == NodeType::List)

70 x.constList()[1].constList().end(),

131 std::vector<Import> ImportSolver::parseImport(const std::filesystem::path& file, const Import& import)
132 {
133 const auto path = findFile(file, import);
134 if (path.extension() == ".arkm") // Nothing to import in case of modules
135 {
136 // Creating an import node that will stay there when visiting the AST and
137 // replacing the imports with their parsed module
138 auto module_node = Node(NodeType::List);
139 module_node.push_back(Node(Keyword::Import));
140
141 auto package_node = Node(NodeType::List);
142 for (const std::string& stem : import.package)
143 package_node.push_back(Node(NodeType::String, stem));
144 module_node.push_back(package_node);
145 // empty symbols list
146 module_node.push_back(Node(NodeType::List));
147
148 m_modules[import.toPackageString()] = Module {
149 module_node,
150 true
151 };
152
153 return {};
154 }
155
156 Parser parser;
157 const std::string code = Utils::readFile(path);
158 parser.process(path.string(), code);
159 m_modules[import.toPackageString()] = Module {
160 parser.ast(),
161 false
162 };

176 std::filesystem::path ImportSolver::findFile(const std::filesystem::path& file, const Import& import)
177 {
178 const std::string package_path = import.packageToPath();
179 if (auto maybe_path = testExtensions(m_root, package_path); maybe_path.has_value())
180 return maybe_path.value();
181
182 // search in all folders in environment path
183 for (const auto& path : m_libenv)
184 {
185 if (auto maybe_path = testExtensions(path, package_path); maybe_path.has_value())
186 return maybe_path.value();
187 }
188
189 // fallback, we couldn't find the file
