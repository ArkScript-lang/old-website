135 std::string Node::repr() const noexcept
136 {
137 std::string data;
138 switch (m_type)
139 {
140 case NodeType::Symbol:
141 data += string();
142 break;
143
144 case NodeType::Capture:
145 data += "&" + string();
146 break;
147
148 case NodeType::Keyword:
149 data += keywords[static_cast<std::size_t>(keyword())];
150 break;
151
152 case NodeType::String:
153 data += "\"" + string() + "\"";
154 break;
155
156 case NodeType::Number:
157 data += fmt::format("{}", number());
158 break;
159
160 case NodeType::List:
161 data += "(";
162 for (std::size_t i = 0, end = constList().size(); i < end; ++i)
163 {
164 data += constList()[i].repr();
165 if (i < end - 1)
166 data += " ";
167 }
168 data += ")";
169 break;
170
171 case NodeType::Field:
172 for (std::size_t i = 0, end = constList().size(); i < end; ++i)
173 {
174 data += constList()[i].repr();
175 if (i < end - 1)
176 data += ".";
177 }
178 break;
179
180 case NodeType::Macro:
181 data += "($ ";
182 for (std::size_t i = 0, end = constList().size(); i < end; ++i)
183 {
184 data += constList()[i].repr();
185 if (i < end - 1)
186 data += " ";
187 }
188 data += ")";
189 break;
190
191 case NodeType::Spread:
192 data += "..." + string();
193 break;
194
195 case NodeType::Unused:
196 break;
197 }
198 return data;
